main() {
    int f;
    int i = 1;
    println("Ingrese el n√∫mero de filas");
    input(f);
    while(i <= f) {
        int espacio = f - i;
        int j = 0;
        /*Imprime los espacios antes de los asteriscos*/
        while(j < espacio) {
            print(" ");
            j++;
        }
        int k = 0;
        /*Imprime los asteriscos de la fila actual*/
        while(k < (2 * i - 1)) {
            print("*");
            k++;
        }
        println("");
        i++;
    }
}
